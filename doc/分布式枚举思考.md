# 分布式枚举

下面是我初步尝试考虑的一种分布式枚举方案。这个方案淋漓尽致地表现出了这个实现的问题：

这里的`main.zig`使用的是原版`cold_color`，而`test_color.zig`使用的是`extend_cold_color`，而`main.zig`又使用了`test_color.zig`的函数。那么这里的处理方法是：

`main.zig`中直接打印的内容，是`"unknown color"`。而调用`testColor`中的打印是正常的。

这个处理会导致代码膨胀：对于扩展和未扩展的分布，事实上形成两套不同的处理逻辑。

这意味着，`Color`本质成为了一个泛型，而不再是一个唯一的确定的分布式枚举，所有使用`c: Color`为参数的函数，都有可能膨胀。

如果实际实现的时候，对于运行时的内容采用了两套impl，会造成确定的代码膨胀。主要槽点是，越写越像接口，而不像分布式enum了。

``` zig
// color.zig
pub const Color = distributed enum {
    red = 0,
    yellow = 1,
    distributing,
};
pub fn draw(c: Color) void {
    switch (c) {
        .red => drawRed(),

const Color = @import("color.zig").Color;

    switch (c) {
        .red => "red",
        .yellow => "yellow",
        distributing [[color_name_switch]],
        else => "unknown color";
    }
}
// cold_color.zig
pub const cold_color = @import("color.zig").Color.rootDistribute(.{
    green = 2,
    blue = 3,
}).distribute(@import("color.zig").[[draw_switch]], .{
    .green => drawGreen(),
    .blue => drawBlue(),
});
// cold_color_extend.zig
pub const cold_color = @import("cold_color.zig").distribute( @import("color_name.zig").[[color_name_switch]], .{
    .green => "green",
    .blue => "blue",
});
// test_color.zig
const Color = @import("color.zig").Color;
const cold_color = @import("cold_color_extend.zig").cold_color;
const colorName = @import("color_name.zig").colorName;
pub fn testColor() void {
    cosnt TestColor = Color.impl(.{cold_color});
    const c = TestColor.green;
    std.debug.print("{s}", .{colorName(c)});
}
// main.zig
const Color = @import("color.zig").Color;
const cold_color = @import("cold_color.zig").cold_color;
const colorName = @import("color_name.zig").colorName;
const testColor = @import("test_color.zig").testColor;
pub fn main(c: Color) {
    const MainColor = Color.impl(.{cold_color});
    const c = MainColor.green;
    std.debug.print("{s}", .{colorName(c)});
    testColor();
};
```

经过反思——这个语法是无意义的，我完全可以用目前的元编程实现这样的库：

``` zig
// distro_enum.zig
pub fn define(comptime options: struct {
    base_fields: anytype,
    switch_slots: []const []const u8,
}) type {
    return struct {
        pub const base_fields = options.base_fields;
        pub const switch_slots = options.switch_slots;
    };
}
pub fn extend(comptime template: type, comptime options: anytype) type {
    // ... 在编译期检查 options 提供的字段名是否与 template.switch_slots 匹配 ...
    return options; // options 结构体本身就包含了所有需要的信息
}
// stitch() 现在返回一个“上下文”结构体
pub fn stitch(comptime template: type, comptime extensions: anytype) type {
    return struct {
        // ✨ 1. 公开最终的、缝合好的 enum 类型 ✨
        pub const Enum = comptime blk: {
            // ... 使用 @Type 从 template 和 extensions 生成最终的 enum ...
        };

        // ✨ 2. 提供一个“分支生成器”工具函数 ✨
        //    这个函数是库的核心价值所在
        pub fn getBranches(comptime slot_name: []const u8) anytype {
            // 在编译期，遍历所有 extensions，
            // 找到与 slot_name 匹配的分支数据，
            // 然后返回一个包含所有这些分支的 comptime 元组。
            // ...
        }
    };
}
// color_core.zig
const distro = @import("distro_enum.zig");

pub const ColorTemplate = distro.define(.{
    .base_fields = .{ .red = 0, .yellow = 1 },
    .switch_slots = &.{ "draw", "colorName" },
});
// cold_color_ext.zig
const distro = @import("distro_enum.zig");
const ColorTemplate = @import("color_core.zig").ColorTemplate;

pub const cold_color_extension = distro.extend(ColorTemplate, .{
    .new_fields = .{ .green = 2, .blue = 3 },
    
    .draw = .{ // 对应 "draw" 插槽
        .{ .tag = .green, .func = drawGreen },
        .{ .tag = .blue, .func = drawBlue },
    },
    .colorName = .{ // 对应 "colorName" 插槽
        .{ .tag = .green, .str = "green" },
        .{ .tag = .blue, .str = "blue" },
    },
});
// my_color_system.zig
const distro = @import("distro_enum.zig");
const ColorTemplate = @import("color_core.zig").ColorTemplate;
const cold_ext = @import("cold_color_ext.zig").cold_color_extension;
// ... import other extensions ...

// 1. ✨ “缝合”所有扩展，得到一个“上下文”对象 ✨
const StitchedColor = distro.stitch(ColorTemplate, .{ cold_ext });

// 2. ✨ 从上下文中，获取最终的 enum 类型 ✨
pub const Color = StitchedColor.Enum;

// 3. ✨ 用户自己定义最终的方法，并“使用”库提供的工具 ✨
pub fn draw(c: Color) void {
    switch (c) {
        // a. 用户可以自由地提供自己的基础/默认分支
        .red => drawRed(),
        .yellow => drawYellow(),
        
        // b. ✨ 调用上下文提供的“分支生成器”来自动填充其余分支 ✨
        //    这行代码清晰地表达了“剩下的分支由扩展模块提供”
        inline for (StitchedColor.getBranches("draw")) |branch| {
            branch.tag => branch.func(),
        },
    }
}

pub fn colorName(c: Color) []const u8 {
    return switch (c) {
        .red => "red",
        .yellow => "yellow",
        inline for (StitchedColor.getBranches("colorName")) |branch| {
            branch.tag => branch.str,
        },
    };
}
```
